005package com.titaniumallies.frcsmartlibrary.comms;
006
007import com.sun.squawk.io.BufferedReader;
008import com.sun.squawk.io.BufferedWriter;
009import java.io.IOException;
010import java.io.InputStreamReader;
011import java.io.OutputStreamWriter;
012import javax.microedition.io.SocketConnection;
013
014/**
015 *
016 * @author Nathanial Lattimer <Nathanial.Lattimer at iismathwizard.com>
017 */
018public class SocketHandler {
019    private SocketConnection _sock;
020    private boolean _isConnected = false;
021    
022    public SocketHandler(SocketConnection sock)
023    {
024        this._sock = sock;
025        _isConnected = true;
026    }
027    
028    public String read()
029    {
030        BufferedReader input;
031        String returnValue = null;
032        try {
033            input = new BufferedReader(new InputStreamReader(_sock.openInputStream()));
034            returnValue = input.readLine();
035            input.close();
036        } catch (IOException ex) {
037            
038        }
039        
040        return returnValue;
041    }
042    
043    public boolean write(String msg)
044    {
045        BufferedWriter writer;
046        try {
047            writer = new BufferedWriter(new OutputStreamWriter(_sock.openOutputStream()));
048            writer.write(msg);
049            writer.flush();
050            writer.close();
051            return true;
052        } catch (IOException ex) {
053            return false;
054        }
055        catch(Exception e)
056        {
057            return false;
058        }
059    }
060    
061    public boolean isConnected()
062    {
063        return this._isConnected;
064    }
065}
