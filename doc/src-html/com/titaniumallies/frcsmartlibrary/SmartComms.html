007import edu.wpi.first.wpilibj.Encoder;
008import edu.wpi.first.wpilibj.Joystick;
009import edu.wpi.first.wpilibj.SpeedController;
010import edu.wpi.first.wpilibj.camera.AxisCamera;
011import edu.wpi.first.wpilibj.networktables.NetworkTable;
012import edu.wpi.first.wpilibj.networktables2.util.List;
013
014/**
015 * This class allows for communicating vital information to other machines
016 * @author Nathanial Lattimer <Nathanial.Lattimer at iismathwizard.com>
017 */
018public class SmartComms {
019    /*
020     * Vital information would be:
021     *      Motor outputs (set value/ encoder feedback)
022     *          Perceived speed
023     *      Joystick inputs
024     *      Camera
025     *      
026     */
027    private static List joys = new List();
028    private static List mtrEncPairs = new List();
029    private static AxisCamera cam = null;
030    
031    public static void addJoystick(Joystick joy)
032    {
033        joys.add(joy);
034    }
035    
036    public static void addMotorEncPair(SpeedController mtr, Encoder enc)
037    {
038        MotorEncoderPair pair = new MotorEncoderPair(mtr, enc);
039        mtrEncPairs.add(pair);
040    }
041    
042    public static void registerCamera(AxisCamera camera)
043    {
044        cam = camera;
045    }
046    
047    public static void start()
048    {
049        NetworkTable table = NetworkTable.getTable("RoboComms");
050        
051    }
052    
053    private static class MotorEncoderPair
054    {
055        SpeedController mtr;
056        Encoder enc;
057        
058        public MotorEncoderPair(SpeedController mtr, Encoder enc)
059        {
060            this.mtr = mtr;
061            this.enc = enc;
062        }
063    }
064}
